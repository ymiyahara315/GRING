function App() {

const [selectedPlan, setSelectedPlan] = React.useState(null);

const [selectedOptions, setSelectedOptions] = React.useState([]);

const [selectedLocation, setSelectedLocation] = React.useState('');

const [copySuccess, setCopySuccess] = React.useState(false);

// オールインパッケージプラン選択時に自動選択するオプション

React.useEffect(() => {

if (selectedPlan === 'all-in') {

// 自動選択されるオプションのIDを設定

const autoSelectedOptions = ['ishibashi-shimei', 'bride-rental', 'groom-rental'];

// 既存の選択から自動選択オプションを除外

const filteredOptions = selectedOptions.filter(

optId => !autoSelectedOptions.includes(optId)

);

// 新しい選択に自動選択オプションを追加

setSelectedOptions([...filteredOptions, ...autoSelectedOptions]);

}

}, [selectedPlan]);

const handlePlanSelect = (planId) => {

setSelectedPlan(planId);

};

const handleOptionToggle = (optionId) => {

// オールインパッケージプランで自動選択されるオプションは切り替え不可

if (selectedPlan === 'all-in') {

const autoSelectedOptions = ['ishibashi-shimei', 'bride-rental', 'groom-rental'];

if (autoSelectedOptions.includes(optionId)) {

return;

}

}

setSelectedOptions(prevOptions => {

if (prevOptions.includes(optionId)) {

return prevOptions.filter(id => id !== optionId);

} else {

return [...prevOptions, optionId];

}

});

};

const handleLocationChange = (e) => {

setSelectedLocation(e.target.value);

};

// 料金計算

const calculateTotal = () => {

if (!selectedPlan) return 0;

// 基本プラン料金

const planObj = basePlans.find(plan => plan.id === selectedPlan);

let total = planObj ? planObj.price : 0;

// オプション料金

selectedOptions.forEach(optionId => {

// オールインパッケージプランに含まれるオプションは料金に含めない

if (selectedPlan === 'all-in' && ['ishibashi-shimei', 'bride-rental', 'groom-rental'].includes(optionId)) {

return;

}

const optionObj = options.find(option => option.id === optionId);

if (optionObj) {

total += optionObj.price;

}

});

// ロケーション料金（オールインパッケージプランの場合は含めない）

if (selectedLocation && selectedPlan !== 'all-in') {

const locationObj = locations.find(location => location.id === selectedLocation);

if (locationObj) {

total += locationObj.price;

}

}

return total;

};

// コピー用のテキストを生成

const generateCopyText = () => {

if (!selectedPlan) return '';

const planObj = basePlans.find(plan => plan.id === selectedPlan);

let text = '【選択プラン】\n';

text += `- 基本プラン: ${planObj.name} - ¥${planObj.price.toLocaleString()}\n`;

const selectedOptionsDetails = selectedOptions

.map(optionId => {

const option = options.find(opt => opt.id === optionId);

if (!option) return null;

// オールインパッケージプランに含まれるオプションは料金表示を変更

if (selectedPlan === 'all-in' && ['ishibashi-shimei', 'bride-rental', 'groom-rental'].includes(optionId)) {

return `- ${option.name} - (プランに含む)`;

}

return option.price > 0 ? `- ${option.name} - ¥${option.price.toLocaleString()}` : `- ${option.name}`;

})

.filter(Boolean);

if (selectedOptionsDetails.length > 0) {

text += '- オプション:\n';

text += selectedOptionsDetails.join('\n');

text += '\n';

}

if (selectedLocation) {

const locationObj = locations.find(location => location.id === selectedLocation);

if (locationObj) {

// オールインパッケージプランの場合、ロケーション料金は0円

const locationPrice = selectedPlan === 'all-in' ? 0 : locationObj.price;

const priceText = selectedPlan === 'all-in' ? '(プランに含む)' : `¥${locationPrice.toLocaleString()}`;

text += `- ロケーション: ${locationObj.name} - ${priceText}\n`;

}

}

text += `\n【合計金額】 ¥${calculateTotal().toLocaleString()}`;

return text;

};

// クリップボードにコピー

const copyToClipboard = () => {

const textToCopy = generateCopyText();

navigator.clipboard.writeText(textToCopy).then(() => {

setCopySuccess(true);

setTimeout(() => {

setCopySuccess(false);

}, 2000);

});

};

return (

<div className="container">

<div className="header">

<h1>GRING自動お見積りツール</h1>

</div>

<div className="grid-container">

<div className="form-section">

<div className="form-container">

<h2 className="section-title">基本プラン</h2>

<div className="radio-group">

{basePlans.map(plan => (

key={plan.id}

>

<input

type="radio"

name="plan"

className="radio-input"

checked={selectedPlan === plan.id}

onChange={() => handlePlanSelect(plan.id)}

/>

<div className="plan-details">

<div className="plan-name">{plan.name}</div>

<div className="plan-price">¥{plan.price.toLocaleString()}</div>

{plan.description && (

<div className="plan-description">{plan.description}</div>

)}

</div>

</label>

))}

</div>

</div>

{selectedPlan && (

<React.Fragment>

<div className="form-container">

<h2 className="section-title">オプション</h2>

<div className="checkbox-group">

{options.map(option => {

// オールインパッケージプランの場合に自動選択されるオプションかチェック

const isAutoSelected = selectedPlan === 'all-in' &&

['ishibashi-shimei', 'bride-rental', 'groom-rental'].includes(option.id);

return (

<label

key={option.id}

className={`checkbox-label ${selectedOptions.includes(option.id) ? 'selected' : ''}`}

>

<input

type="checkbox"

className="checkbox-input"

checked={selectedOptions.includes(option.id)}

onChange={() => handleOptionToggle(option.id)}

disabled={isAutoSelected}

/>

<div className="plan-details">

<div className="plan-name">{option.name}</div>

{option.price > 0 && (

<div className="plan-price">¥{option.price.toLocaleString()}</div>

)}

{option.description && (

<div className="plan-description">{option.description}</div>

)}

</div>

</label>

);

})}

</div>

</div>

<div className="form-container">

<h2 className="section-title">ロケーション</h2>

<select

className="location-select"

value={selectedLocation}

onChange={handleLocationChange}

>

<option value="">ロケーションを選択してください</option>

{locations.map(location => (

<option key={location.id} value={location.id}>

{location.name} - ¥{location.price.toLocaleString()}

{location.description ? ` (${location.description})` : ''}

</option>

))}

</select>

</div>

</React.Fragment>

)}

</div>

<div className="estimation-result">

<h2 className="result-title">見積もり結果</h2>

{selectedPlan ? (

<React.Fragment>

<div className="result-list">

{/* 基本プラン */}

<div className="result-item">

<span className="result-item-name">

{basePlans.find(plan => plan.id === selectedPlan)?.name}

</span>

<span className="result-item-price">

¥{basePlans.find(plan => plan.id === selectedPlan)?.price.toLocaleString()}

</span>

</div>

{/* 選択されたオプション */}

{selectedOptions.map(optionId => {

const option = options.find(opt => opt.id === optionId);

// オールインパッケージプランに含まれるオプションは表示を変更

const isIncludedInAllIn = selectedPlan === 'all-in' && ['ishibashi-shimei', 'bride-rental', 'groom-rental'].includes(optionId);

return (

<div className="result-item" key={optionId}>

<span className="result-item-name">{option?.name}</span>

<span className="result-item-price">

{isIncludedInAllIn ? '(プランに含む)' : (option?.price > 0 ? `¥${option?.price.toLocaleString()}` : '')}

</span>

</div>

);

})}

{/* 選択されたロケーション */}

{selectedLocation && (

<div className="result-item">

<span className="result-item-name">

ロケーション: {locations.find(loc => loc.id === selectedLocation)?.name}

</span>

<span className="result-item-price">

{selectedPlan === 'all-in'

? '(プランに含む)'

: `¥${locations.find(loc => loc.id === selectedLocation)?.price.toLocaleString()}`}

</span>

</div>

)}

</div>

<div className="total">

<span>合計金額</span>

<span>¥{calculateTotal().toLocaleString()}</span>

</div>

<div className="copy-btn-container">

<button className="copy-button" onClick={copyToClipboard}>

見積もりをコピー

</button>

<span className="copy-instruction">

ご依頼ご相談の際はこちらのコピーを公式ラインでお送りください。

</span>

</div>

{copySuccess && <div className="copy-message">コピーしました！</div>}

</React.Fragment>

) : (

<p>基本プランを選択してください。</p>

)}

</div>

</div>

</div>

);

}
